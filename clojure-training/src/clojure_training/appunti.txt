Appunti per la creazione dell'array di set
i set conterranno map unique
;;dichiaro array vuoto
(def barray [])
;;aggiungo all'array un elemento di tipo set
;;il cast a vector del map sembra risolvere il problema di ieri
(def barray (conj barray (into #{} (vector {:a 1 :b 3}) )))
;;aggiungo un altro set all'array
(def barray (conj barray (into #{} (vector {:a 1 :b 22}) )))
;;aggiungo nel set 1 dell'array un'altra mappa
(def barray (assoc barray 1 (set/union (get barray 1) (vector {:a 1 :b 223}) )))
;;risultato della stampa 
[#{{:b 3, :a 1}} #{{:b 22, :a 1} {:b 223, :a 1}}]
